Text
net.sf.tapestry.form.Text
Text Input:
Description
Used to implement a textarea form element (<textarea cols="40" rows="5" value=" ..... ">). Must be wrapped by a Form component.
See Also
Form, TextField
Parameters    

Informal parameters are allowed. The component may not contain a body.
Parameter Type Direction Required Default Description
value java.lang.String in-out no post The text inside the textarea. The parameter is only updated when the the Text component is not disabled.
disabled boolean in no false Controls whether the textarea is active or not. If disabled, then any value that comes up when the form is submitted is ignored.

Corresponds to the disabled HTML attribute.
columns integer in no   The width of the textarea, in characters. If zero, or unspecified the value is left to the client browser to determine.

Corresponds to the cols HTML attribute.
rows integer in no   The number of rows in the textarea. If unspecified or zero, then the value is left to the client browser to determine.

Corresponds to the maxlength HTML attribute.

Examples

Provides a checkbox for the user contact a sales representative.

import net.sf.tapestry.IRequestCycle;
import net.sf.tapestry.html.BasePage;

/**
 * Simply demonstration of Text component
 * @author Neil Clayton
 */
public class TextArea extends BasePage {
	public void formSubmit(IRequestCycle requestCycle) {
	}

	/**
	 * @see net.sf.tapestry.AbstractPage#detach()
	 */
	public void detach() {
		columns = 0;
		rows = 0;
		value = null;
	}

	/**
	 * Returns the columns.
	 * @return int
	 */
	public int getColumns() {
		return columns;
	}

	/**
	 * Returns the rows.
	 * @return int
	 */
	public int getRows() {
		return rows;
	}

	/**
	 * Returns the value.
	 * @return String
	 */
	public String getValue() {
		return value;
	}

	/**
	 * Sets the columns.
	 * @param columns The columns to set
	 */
	public void setColumns(int columns) {
		this.columns = columns;
		fireObservedChange("columns", columns);
	}

	/**
	 * Sets the rows.
	 * @param rows The rows to set
	 */
	public void setRows(int rows) {
		this.rows = rows;
		fireObservedChange("rows", rows);
	}

	/**
	 * Sets the value.
	 * @param value The value to set
	 */
	public void setValue(String value) {
		this.value = value;
		fireObservedChange("value", value);
	}

	private int columns = 20;
	private int rows = 5;
	private String value = "TextArea's intial demonstration value";
}
